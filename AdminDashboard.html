        function handleModalVerification(transactionId, newStatus) {

            const card = document.getElementById('modal-tx-' + transactionId);

            const buttons = card.querySelectorAll('button');

            

            // Disable buttons dan show loading

            buttons.forEach(function(btn) {

                btn.disabled = true;

                btn.style.opacity = '0.5';

            });

            

            const payload = { transactionId: transactionId, newStatus: newStatus };

            

            google.script.run

                .withSuccessHandler(function(response) {

                    if (response.success) {

                        // Update status badge in modal

                        const statusBadge = newStatus === 'Approved' ? 

                            '<span class="px-3 py-1 text-xs font-semibold rounded-full bg-green-100 text-green-800">Disetujui</span>' :

                            '<span class="px-3 py-1 text-xs font-semibold rounded-full bg-red-100 text-red-800">Ditolak</span>';

                        

                        const statusContainer = card.querySelector('.flex.items-center.space-x-3.mb-2');

                        const currentHtml = statusContainer.innerHTML;

                        const newHtml = currentHtml.replace(/<span class="px-3.*?<\/span>/, statusBadge);

                        statusContainer.innerHTML = newHtml;

                        

                        // Remove action buttons

                        const actionButtons = card.querySelector('.flex.space-x-2.mt-4.pt-4.border-t.border-gray-200');

                        if (actionButtons) {

                            actionButtons.remove();

                        }

                        

                        // Refresh main dashboard data

                        loadAdminData();

                    } else {

                        alert(response.message);

                        // Re-enable buttons

                        buttons.forEach(function(btn) {

                            btn.disabled = false;

                            btn.style.opacity = '1';

                        });

                    }

                })

                    alert('Error: ' + error.message);

                    buttons.forEach(function(btn) {

                        btn.disabled = false;

                        btn.style.opacity = '1';

                    });

                })

                .verifyTransaction(payload);

        }

        function sortTable(colIndex) { 

            const isAsc = sortDirection[colIndex] = !sortDirection[colIndex]; 

            filteredNasabahData.sort(function(a, b) { 

                let valA, valB; 

                switch(colIndex) { 

                    case 0: valA = a.name.toLowerCase(); valB = b.name.toLowerCase(); break; 

                    case 1: valA = a.method.toLowerCase(); valB = b.method.toLowerCase(); break; 

                    case 2: valA = a.totalSavings; valB = b.totalSavings; break; 

                    case 3: valA = a.progress; valB = b.progress; break; 

                    case 4: valA = a.status.toLowerCase(); valB = b.status.toLowerCase(); break; 

                } 

                if (valA < valB) return isAsc ? -1 : 1; 

                if (valA > valB) return isAsc ? 1 : -1; 

                return 0; 

            }); 

            renderTable(filteredNasabahData); 

        }

        

        function filterData() {

            const searchTerm = document.getElementById('searchInput').value.toLowerCase();

            const statusFilter = document.getElementById('statusFilter').value;

            

            filteredNasabahData = allNasabahData.filter(function(user) {

                const matchesSearch = user.name.toLowerCase().indexOf(searchTerm) !== -1 || 

                                    user.email.toLowerCase().indexOf(searchTerm) !== -1;

                const matchesStatus = !statusFilter || user.status === statusFilter;

                

                return matchesSearch && matchesStatus;

            });

            

            renderTable(filteredNasabahData);

        }
