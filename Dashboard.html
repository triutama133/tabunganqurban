        function readFileAsBase64(file) {

            return new Promise((resolve, reject) => {

                const reader = new FileReader();

                reader.onload = () => resolve(reader.result.split(',')[1]);

                reader.onerror = error => reject(error);

                reader.readAsDataURL(file);

            });

        }

        function logout() { localStorage.removeItem('qurbanUser'); window.top.location.href = "<?= ScriptApp.getService().getUrl() ?>?page=login"; }

        

        function showConfirmModal(id) { itemToDeleteId = id; confirmModal.style.display = 'flex'; }

        function hideConfirmModal() { itemToDeleteId = null; confirmModal.style.display = 'none'; }

        

        function handleDelete() {

            if (itemToDeleteId) {

                const deleteButton = document.getElementById('confirmDelete'); const cancelButton = document.getElementById('cancelDelete'); const spinner = document.getElementById('deleteSpinner'); const buttonText = document.getElementById('deleteButtonText');

                deleteButton.disabled = true; cancelButton.disabled = true;

                spinner.classList.remove('hidden'); buttonText.classList.add('hidden');

                document.getElementById('historyLoader').classList.remove('hidden');


                google.script.run.withSuccessHandler(response => {

                    if(response.success) {

                        google.script.run.withSuccessHandler(refreshPartialData).getDashboardData(currentUser.userId);

                    } else { alert(response.message); }

                    spinner.classList.add('hidden'); buttonText.classList.remove('hidden');

                    hideConfirmModal();

                }).withFailureHandler(error => {

                    alert('Error: ' + error.message);

                    deleteButton.disabled = false; cancelButton.disabled = false;

                    spinner.classList.add('hidden'); buttonText.classList.remove('hidden');

                    hideConfirmModal();

                }).deleteSaving({ transactionId: itemToDeleteId, userId: currentUser.userId });

            }
